\begin{document}
	Bode plot in ZPK format
	\begin{center}
		\BodeZPK[red,thick]{{0,{-0.1,-0.5},{-0.1,0.5}},{{-0.1,-10},{-0.1,10}},10}{0.01}{100}
	\end{center}
		
	Bode plot in TF format
	\begin{center}
		\BodeTF[blue,thick]{{10,2,2.6000,0},{1,0.2,100.01}}{0.01}{100}
	\end{center}

	Linear approximation with customization
	\begin{center}
		\BodeZPK[{{red,thick},{blue,thick}},{{ytick distance=40},{ytick distance=90}},linear] {{0,{-0.1,-0.5},{-0.1,0.5}},{{-0.1,-10},{-0.1,10}},10} {0.01} {100}
	\end{center}

	Plot with delay and customization
	\begin{center}
		\BodeZPK[{{red,thick},{blue,thick}},{{ytick distance=40},{ytick distance=90}}] {{0,{-0.1,-0.5},{-0.1,0.5}},{{-0.1,-10},{-0.1,10}},10,0.1} {0.01} {100}
	\end{center}

	Individual gain and phase plots with more customization
	\begin{center}
		\begin{BodePlot}[ylabel={Gain (dB)},
			ytick distance=40, height=2cm, width=4cm]{0.01}{100}
			\addBodeZPKPlots[true/{black,thick},linear/{red,dashed,thick},asymptotic/{blue,dotted,thick}]{magnitude}{{0,{-0.1,-0.5},{-0.1,0.5}},{{-0.1,-10},{-0.1,10}},10}
		\end{BodePlot}
		\begin{BodePlot}[ylabel={Phase ($^{\circ}$)},height=2cm,width=4cm,ytick distance=90,]{0.01}{100}
			\addBodeZPKPlots[true/{black,thick},linear/{red,dashed,thick},asymptotic/{blue,dotted,thick}]{phase}{{0,{-0.1,-0.5},{-0.1,0.5}},{{-0.1,-10},{-0.1,10}},10}
		\end{BodePlot}
	\end{center}
